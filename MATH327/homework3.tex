    \begin{itemize}
        \item $P(T=5) = F(5)-F(4) = \frac{1}{4}$
        \item $P(T>3) = 1-F(3) = \frac{1}{2}$
        \item $P(1.4 < T < 6) = F(6)-F(1.4)=\frac{1}{2}$
        \item $P(T\leq 5 \mid T\geq 2) = \frac{P(2\leq T\leq 5)}{P(T\geq 2)} = \frac{F(5)-F(2)}{1-F(2)} = \frac{2}{3}$
    \end{itemize}
